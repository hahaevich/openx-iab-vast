Example 10 - a mixed playlist

This example shows a mixed playlist which has 2 different show streams (running for 1 minute each) coming from 2 different netconnection URLs with pre-roll and overlay ads (click to show a popup video) being served in each stream (5 seconds into each stream). At the front of the playlist is a splash image. Demonstrates how multiple ad streams can be configured in a mixed playlist.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.0.swf", {
    playlist: [
        {
            url: 'images/splash.jpg'
        },
        {
            url: 'main',
            provider: 'openXAdStreamer'
        },
        {
            url: 'popup',
            provider: 'openXAdStreamer'
        },
        {
            url: 'main',
            provider: 'openXAdStreamer2'
        },
        {
            url: 'popup',
            provider: 'openXAdStreamer2'
        }
    ],

    plugins: {
        openXRegions: {
            url: '../dist/OpenXRegions-0.4.7.swf'    		
        },

        openXAdStreamer: {
            url: '../dist/OpenXAdStreamer-0.4.7.swf',
            netConnectionUrl: 'rtmp://ne7c0nwbit.rtmphost.com/videoplayer',
            streams: [ { file:'mp4:bbb_640x360_h264', duration:'00:01:00' } ],
            vastServerURL: 'http://openx.bouncingminds.com/openx/www/delivery/fc.php',
            regionsPluginName: 'openXRegions',
            debugLevel: 'fatal',
            adSchedule: [ 
                { zone: '1',
                  position: 'pre-roll'
                },
                { zone: '2',
                  position: 'bottom',
                  width: 600,
                  height: 40,
                  startTime: '00:00:05',
                  duration: '15'
                }
            ]
        },
      		
        openXAdStreamer2: {
            url: '../dist/OpenXAdStreamer-0.4.7.swf',
            netConnectionUrl: 'rtmp://ne7c0nwbit.rtmphost.com/videoplayer',
            streams: [ { file:'mp4:john_bailey_intro_600' , duration:'00:01:00' } ],
            vastServerURL: 'http://openx.bouncingminds.com/openx/www/delivery/fc.php',
            regionsPluginName: 'openXRegions',
            debugLevel: 'fatal',
            adSchedule: [ 
                { zone: '1',
                  position: 'pre-roll'
                },
                { zone: '2',
                  position: 'bottom',
                  width: 600,
                  height: 40,
                  startTime: '00:00:05',
                  duration: '15'
                }
            ]
        }      		
    }
});
