Example 13 - Repeating Ad Positions

This example plays 3 ads sequentially 10 seconds into the main stream.


The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.0.swf", {
    playlist: [
        {
            url: 'main',
            provider: 'openXAdStreamer'
        }
    ],
    
    plugins: {
        openXRegions: {
            url: '../dist/OpenXRegions-0.4.6.swf'
        },

        openXAdStreamer: {
            url: '../dist/OpenXAdStreamer-0.4.6.swf',
            netConnectionUrl: 'rtmp://ne7c0nwbit.rtmphost.com/videoplayer',
            streams: [ { file:'mp4:bbb_640x360_h264', duration: '00:09:56' } ],
            vastServerURL: 'http://beta.bouncingminds.com/openx-2.8.1-rc7/www/delivery/fc.php',
            debugLevel: 'fatal',
            allowAdRepetition: true,
            adSchedule: [
                // plays 3 ads sequentially 10 seconds into the main stream
                { zone: '2',
                  position: 'mid-roll',
                  startTime: '00:00:10',
                  repeat: 3
                }
            ]
        }
    }
});
