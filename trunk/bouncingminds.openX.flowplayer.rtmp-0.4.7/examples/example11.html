Example 11 - A live stream with mid-roll video ad

A live stream (to run indefinately) that has a mid-roll video ad played 10 seconds into the stream. Once the live stream resumes, an overlay is shown 15 seconds into the resumed stream for 30 seconds. Please note, the live stream may not be active when you view this example.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.0.swf", {
    playlist: [
        {
            provider: 'liveStream_10secs',
            url: 'main'
        },
        {
            provider: 'midRollAdStreamer',
            url: 'main'
        },
        {
            provider: 'liveStream_continue',
            url: 'main'
        },
        {
            provider: 'liveStream_continue',
            url: 'popup'
        }
    ],

    plugins: {
        openXRegions: {
            url: '../dist/OpenXRegions-0.4.7.swf',
            debugLevel: 'fatal'
        },

        liveStream_10secs: {
            url: '../dist/OpenXAdStreamer-0.4.7.swf',
            netConnectionUrl: 'rtmp://horsetv.fc.llnwd.net/horsetv',
            streams: [ { file:'(live)foalcam', duration: '00:00:10' } ],
            subscribe: true,
            debugLevel: 'fatal'
        },

        midRollAdStreamer: {
            url: '../dist/OpenXAdStreamer-0.4.7.swf',
            netConnectionUrl: 'rtmp://ne7c0nwbit.rtmphost.com/videoplayer',
            vastServerURL: 'http://openx.bouncingminds.com/openx/www/delivery/fc.php',
            debugLevel: 'fatal',
            adSchedule: [
                { zone: '1',
                  position: 'mid-roll'
                }
            ]
        },

        liveStream_continue: {
            url: '../dist/OpenXAdStreamer-0.4.7.swf',
            netConnectionUrl: 'rtmp://horsetv.fc.llnwd.net/horsetv',
            streams: [ { file:'(live)foalcam' } ],
            vastServerURL: 'http://openx.bouncingminds.com/openx/www/delivery/fc.php',
            debugLevel: 'fatal',
            subscribe: true,
            adSchedule: [
                { zone: '4',
                  position: 'bottom',
				  startTime: '00:00:15',
				  duration: '30'
                }
            ]
        }
    }
});
